**2. Create `apps/server/src/routes/channels/unread.ts`:**

`GET /api/servers/:serverId/unread` — Returns unread counts for all channels in a server for the authenticated user.

- Requires auth middleware
- Query: Join channels (where serverId matches) with messages and channelReadStates
- For each channel, count messages where `createdAt > COALESCE(lastReadAt, epoch)` AND `senderId != userId`
- Return: `Array<{ channelId: string; unreadCount: number }>`

Use Drizzle's sql template for the count query:
```typescript
import { sql, eq, and, gt, ne } from "drizzle-orm";

// Get all channels for this server
const serverChannels = await db.select({ id: channels.id })
  .from(channels)
  .where(eq(channels.serverId, serverId));

// For each channel, get the user's lastReadAt
// Then count messages newer than that, excluding own messages
// Do this in a single query with LEFT JOIN and GROUP BY for efficiency

const results = await db
  .select({
    channelId: channels.id,
    unreadCount: sql<number>`
      COUNT(CASE
        WHEN ${messages.createdAt} > COALESCE(${channelReadStates.lastReadAt}, '1970-01-01'::timestamp)
        AND ${messages.senderId} != ${userId}
        THEN 1
      END)
    `.as("unread_count"),
  })
  .from(channels)
  .leftJoin(
    channelReadStates,
    and(
      eq(channelReadStates.channelId, channels.id),
      eq(channelReadStates.userId, userId),
    ),
  )
  .leftJoin(messages, eq(messages.channelId, channels.id))
  .where(eq(channels.serverId, serverId))
  .groupBy(channels.id, channelReadStates.lastReadAt);
```

Return the results as JSON.

**1. Create `apps/client/src/hooks/useUnread.ts`:**

Uses TanStack Query to fetch unread counts from the server, and Socket.IO events for real-time updates.

```typescript
import { useQuery, useQueryClient } from "@tanstack/react-query";
import { useCallback } from "react";
import { useSocket } from "./useSocket";
import { apiFetch } from "@/lib/api";

interface ChannelUnread {
  channelId: string;
  unreadCount: number;
  hasMention: boolean; // Set client-side based on mention detection
}

// Fetch unread counts for all channels in a server
export function useUnread(serverId: string | undefined) {
  return useQuery<ChannelUnread[]>({
    queryKey: ["unread", serverId],
    queryFn: async () => {
      if (!serverId) return [];
      const res = await apiFetch(`/api/servers/${serverId}/unread`);
      if (!res.ok) throw new Error("Failed to fetch unread counts");
      const data = await res.json();
      // Server returns { channelId, unreadCount }[], add hasMention: false (detected client-side)
      return data.map((d: { channelId: string; unreadCount: number }) => ({
        ...d,
        unreadCount: Number(d.unreadCount),
        hasMention: false,
      }));
    },
    enabled: !!serverId,
    refetchInterval: 60_000, // Refresh every 60 seconds as fallback
  });
}

// Get unread count for a specific channel
export function useChannelUnread(serverId: string | undefined, channelId: string) {
  const { data: unreads } = useUnread(serverId);
  const entry = unreads?.find((u) => u.channelId === channelId);
  return {
    unreadCount: entry?.unreadCount ?? 0,
    hasMention: entry?.hasMention ?? false,
  };
}

// Check if any channel in a server has unreads (for server icon badge)
export function useServerHasUnread(serverId: string | undefined) {
  const { data: unreads } = useUnread(serverId);
  const totalUnread = unreads?.reduce((sum, u) => sum + u.unreadCount, 0) ?? 0;
  const hasMention = unreads?.some((u) => u.hasMention) ?? false;
  return { totalUnread, hasMention };
}

// Mark a channel as read (via socket event)
export function useMarkChannelRead() {
  const socket = useSocket();
  const queryClient = useQueryClient();

  return useCallback(
    (channelId: string, serverId: string) => {
      socket.emit("channel:read", { channelId });
      // Optimistically clear the unread count in the cache
      queryClient.setQueryData<ChannelUnread[]>(
        ["unread", serverId],
        (old) =>
          old?.map((u) =>
            u.channelId === channelId
              ? { ...u, unreadCount: 0, hasMention: false }
              : u,
          ) ?? [],
      );
    },
    [socket, queryClient],
  );
}
```
