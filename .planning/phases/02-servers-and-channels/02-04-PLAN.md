                  sql`WHEN ${channels.id} = ${ch.id} THEN ${i}`

  <action>
    Create `apps/client/src/hooks/useChannels.ts` with the following hooks:

    ```typescript
    import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
    import { api } from "@/lib/api";
    import type { ChannelResponse } from "@tether/shared";

    export function useChannels(serverId: string | undefined) {
      return useQuery({
        queryKey: ["servers", serverId, "channels"],
        queryFn: () => api.get<ChannelResponse[]>(`/api/servers/${serverId}/channels`),
        enabled: !!serverId,
      });
    }

    export function useCreateChannel(serverId: string) {
      const queryClient = useQueryClient();
      return useMutation({
        mutationFn: (data: { name: string; type?: "text" | "voice" }) =>
          api.post<ChannelResponse>(`/api/servers/${serverId}/channels`, data),
        onSuccess: () => {
          queryClient.invalidateQueries({ queryKey: ["servers", serverId, "channels"] });
        },
      });
    }

    export function useUpdateChannel(serverId: string) {
      const queryClient = useQueryClient();
      return useMutation({
        mutationFn: ({ channelId, ...data }: { channelId: string; name?: string; topic?: string | null }) =>
          api.patch<ChannelResponse>(`/api/channels/${channelId}`, data),
        onSuccess: () => {
          queryClient.invalidateQueries({ queryKey: ["servers", serverId, "channels"] });
        },
      });
    }

    export function useDeleteChannel(serverId: string) {
      const queryClient = useQueryClient();
      return useMutation({
        mutationFn: (channelId: string) =>
          api.delete(`/api/channels/${channelId}`),
        onSuccess: () => {
          queryClient.invalidateQueries({ queryKey: ["servers", serverId, "channels"] });
        },
      });
    }

    export function useReorderChannels(serverId: string) {
      const queryClient = useQueryClient();
      return useMutation({
        mutationFn: (order: { id: string; position: number }[]) =>
          api.patch(`/api/servers/${serverId}/channels/reorder`, { order }),
        onSuccess: () => {
          queryClient.invalidateQueries({ queryKey: ["servers", serverId, "channels"] });
        },
      });
    }
    ```
