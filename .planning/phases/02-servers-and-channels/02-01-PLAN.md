       ```typescript
       const memberships = await db
         .select({ serverId: serverMembers.serverId })
         .from(serverMembers)
         .where(eq(serverMembers.userId, userId));
       ```

    2. Join personal room and all server rooms:
       ```typescript
       await socket.join(`user:${userId}`);
       for (const { serverId } of memberships) {
         await socket.join(`server:${serverId}`);
       }
       logger.info({ userId, rooms: memberships.length }, "User joined rooms");
       ```

    3. Add a `server:subscribe` event handler for post-join room addition (when user joins a server via invite, their existing socket needs to join the new server room without reconnecting):
       ```typescript
       socket.on("server:subscribe", async ({ serverId }: { serverId: string }) => {
         // Verify the user is actually a member before joining the room
         const [membership] = await db
           .select({ id: serverMembers.id })
           .from(serverMembers)
           .where(and(eq(serverMembers.serverId, serverId), eq(serverMembers.userId, userId)));
         if (membership) {
           await socket.join(`server:${serverId}`);
           logger.info({ userId, serverId }, "User subscribed to server room");
         }
       });
       ```

    4. Add a `server:unsubscribe` event handler (for when a user leaves a server):
       ```typescript
       socket.on("server:unsubscribe", ({ serverId }: { serverId: string }) => {
         socket.leave(`server:${serverId}`);
         logger.info({ userId, serverId }, "User unsubscribed from server room");
       });
       ```

    5. Keep existing `ping/pong` and `disconnect` handlers intact.

    Import `db` from `../../db/client.js`, `serverMembers` from `../../db/schema.js`, `eq` and `and` from `drizzle-orm`.

Update the signature from `function registerConnectionHandlers(...)` to `async function registerConnectionHandlers(...)`. The caller in `socket/index.ts` already does not await it (fire-and-forget from the `io.on("connection")` handler), which is fine for this pattern — but add `.catch(err => logger.error({ err }, "Connection handler error"))` to the call site in `socket/index.ts` to avoid unhandled promise rejections.
