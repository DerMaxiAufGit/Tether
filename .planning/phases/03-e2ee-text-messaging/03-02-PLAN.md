export const messages = pgTable("messages", {
  id: uuid("id").primaryKey().defaultRandom(),
  channelId: uuid("channel_id").notNull().references(() => channels.id, { onDelete: "cascade" }),
  senderId: uuid("sender_id").notNull().references(() => users.id),
  encryptedContent: bytea("encrypted_content").notNull(),
  contentIv: bytea("content_iv").notNull(),
  contentAlgorithm: text("content_algorithm").notNull().default("aes-256-gcm"),
  epoch: integer("epoch").notNull().default(1),
  createdAt: timestamp("created_at").defaultNow().notNull(),
  editedAt: timestamp("edited_at"),
});

export const messageRecipientKeys = pgTable("message_recipient_keys", {
  id: uuid("id").primaryKey().defaultRandom(),
  messageId: uuid("message_id").notNull().references(() => messages.id, { onDelete: "cascade" }),
  recipientUserId: uuid("recipient_user_id").notNull().references(() => users.id),
  encryptedMessageKey: bytea("encrypted_message_key").notNull(),
  ephemeralPublicKey: bytea("ephemeral_public_key").notNull(),

**1. Create `apps/server/src/routes/messages/create.ts`** — POST /api/channels/:channelId/messages

- Auth required (use existing `authenticate` preHandler)
- Validate request body matches `SendMessageRequest` shape: `{ encryptedContent, contentIv, contentAlgorithm?, epoch?, recipients: [{ recipientUserId, encryptedMessageKey, ephemeralPublicKey }] }`
- Verify the authenticated user is a member of the server that owns this channel (JOIN serverMembers on channels.serverId)
- Verify the channel exists and is type "text" (not "voice")
- In a transaction:
  - INSERT into `messages` table: channelId, senderId (from auth), encryptedContent (base64 decode to Buffer), contentIv (base64 decode to Buffer), contentAlgorithm, epoch
  - INSERT N rows into `messageRecipientKeys`: one per recipient with messageId, recipientUserId, encryptedMessageKey (base64 decode to Buffer), ephemeralPublicKey (base64 decode to Buffer)
- After transaction, broadcast `message:created` via Socket.IO to `channel:${channelId}` room. The broadcast payload is a `MessageEnvelope` with all fields as base64 strings (not Buffers). Use `request.server.io.to(...)` since REST handler doesn't have the sender's socket reference. The client will deduplicate via optimistic ID matching.
- Return 201 with the created message in `MessageResponse` format (include the sender's own recipientKey)
- Look up sender's displayName and avatarUrl from users table for the response

**2. Create `apps/server/src/routes/messages/list.ts`** — GET /api/channels/:channelId/messages

- Auth required
- Verify membership (same as create)
- Query params: `before` (optional message ID cursor), `limit` (optional, default 50, max 100)
- SELECT from messages WHERE channelId = :channelId, ordered by createdAt DESC
- If `before` is provided: WHERE createdAt < (SELECT createdAt FROM messages WHERE id = :before)
- JOIN users ON messages.senderId = users.id to get senderDisplayName, senderAvatarUrl
- LEFT JOIN messageRecipientKeys WHERE recipientUserId = authenticated user's ID — this gets only the current user's wrapped key for each message
- Return `{ messages: MessageResponse[] }` — each message includes its recipientKey (the current user's key data), encryptedContent/contentIv as base64 strings
- LIMIT to the requested limit

**3. Create `apps/server/src/routes/messages/delete.ts`** — DELETE /api/messages/:messageId

- Auth required
- Look up the message by ID
- Verify senderId matches the authenticated user (only own messages can be deleted)
- Look up the channelId and serverId for broadcast
- DELETE the message (cascade deletes recipient keys)
- Broadcast `message:deleted` via Socket.IO to `channel:${channelId}` with `{ messageId, channelId }`
- Return 200 with `{ success: true }`
